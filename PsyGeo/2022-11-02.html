# PsychoGeography: {{< meta date >}}

This was my third attempt at making a psychogeography tool for drawing nice routes on the city.

In the first two attempts I was focusing on the mapping side, seing if I could generate the routes from text. This time I decided to try it freestyle in order to gather data about eg. how long it takes to trace the path of a word, how good the result looks on the map, how much doubling back you have to do etc.

```{python}
from ipyleaflet import Map, Marker, basemaps, basemap_to_tiles
import datetime
import os
import json
import pandas as pd
import numpy as np

date = "2022-11-02"

# Import the csv
tmp_root = "./OSM/traces/"
df = pd.read_csv(tmp_root + date + ".csv")

# Check the dates are on the same day
df['date'] = pd.to_datetime(df['date'])
assert df['date'].dt.date.nunique() == 1, "More than one day in the data"

# And that that is the same as the 'date' variable
assert df['date'].dt.date.unique()[0] == datetime.datetime.strptime(date, "%Y-%m-%d").date(), "Date in data does not match date variable"

# Download the runs
import requests
import gpxpy

# Get one trace (gpx)
def get_trace(url):
    r = requests.get(url)
    gpx = gpxpy.parse(r.text)
    return gpx

# Get data from one trace
def get_trace_data(gpx):
    data = []
    for track in gpx.tracks:
        for segment in track.segments:
            for point in segment.points:
                data.append([point.latitude, point.longitude, point.elevation, point.time])
    return pd.DataFrame(data, columns=['lat', 'lon', 'elevation', 'time'])

# Get data from all traces
def get_all_trace_data(run_urls):
    data = []
    for url in run_urls:
        gpx = get_trace(url)
        data.append(get_trace_data(gpx))
    return data

# Get data from all traces
url_df = df
data = get_all_trace_data(url_df['url'])
# Sort by time
data = sorted(data, key=lambda x: x['time'].iloc[0])

for df in data:
    df["time"] = df["time"].dt.tz_convert("Europe/Prague")
    df["time"] = df["time"].dt.tz_localize(None)

# Check the end of each run is not within 1 hour of the start of the next
# If they are, combine them
for i in range(len(data) - 1):
    if data[i] is None:
        continue
    if (data[i]['time'].iloc[-1] - data[i+1]['time'].iloc[0]).total_seconds() < 3600:
        data[i] = pd.concat([data[i], data[i+1]])
        data[i+1] = None

# Remove the None values
data = [x for x in data if x is not None]

# Print a summary
from IPython.display import display, Markdown
display(Markdown(f"There {'were' if len(data) > 1 else 'was'} {len(data)} run{'s' if len(data) > 1 else ''} today."))
for i in range(len(data)):
    start_time = data[i]['time'].iloc[0].strftime("%H:%M")
    end_time = data[i]['time'].iloc[-1].strftime("%H:%M")
    duration = (data[i]['time'].iloc[-1] - data[i]['time'].iloc[0]).total_seconds() / 60
    display(Markdown(f"PsychoGeography {i+1} started at {start_time} and finished at {end_time} ({duration:.1f} minutes)."))

```

```{python}
#| column: screen

def display_run(df):
    # Find the center of the map
    center = [df["lat"].mean(), df["lon"].mean()]

    # Find the bounds
    bounds = [[df["lat"].min(), df["lon"].min()], [df["lat"].max(), df["lon"].max()]]

    # Create the map
    m = Map(center=center, zoom=14)

    # Add line
    from ipyleaflet import Polyline
    line = Polyline(
        locations=df[["lat", "lon"]].values.tolist(),
        color="red",
        fill=False,
    )
    m.add_layer(line)

    # Zoom to line
    m.fit_bounds(bounds)
    return m

for i in range(len(data)):
    display(display_run(data[i]))

```

I feel like it was moderately successful. I think the key to getting really good results will be to write some code that looks all over the city for districts that have shapes that match what you want to draw.

In terms of the psychogeography of it, I turned off my podcast as I got to the top of the first letter, and tried to focus on being in the city as I toured the letters. I tried to keep an awareness both of the letter I was forming, and how it sat in between the buildings, and of the district I was travelling through, and its characteristic features[^1].

I think with practice it should be possible to clean up the letters a bit by thinking more of the calligraphy as you are drawing-- though this would mean more time spent on roads etc; would probably be easier to get better shapes at night when you don't have to stick to the pavements so much.
