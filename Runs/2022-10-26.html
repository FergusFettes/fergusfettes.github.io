```{python}
#| column: screen
from ipyleaflet import Map, Marker, basemaps, basemap_to_tiles

# Load data from gpx
file = '/p/my-quarto-site/_data/run_2022_26_10.gpx'
with open(file, 'r') as f:
    data = f.read()

# Extract coordinates
import re
coords = re.findall(r'<trkpt lat="(.*)" lon="(.*)">', data)
coords = [(float(lat), float(lon)) for lat, lon in coords]

# Find the center of the map
center = (sum([lat for lat, lon in coords]) / len(coords),
          sum([lon for lat, lon in coords]) / len(coords))

# Find the bounds
bounds = ((min([lat for lat, lon in coords]), min([lon for lat, lon in coords])),
          (max([lat for lat, lon in coords]), max([lon for lat, lon in coords])))

m = Map(
    center=center,
    zoom=14,
)

# Add line
from ipyleaflet import Polyline
line = Polyline(
    locations=coords,
    color="red",
    fill=False,
)
m.add_layer(line)

# Zoom to line
m.fit_bounds(bounds)
m
```
